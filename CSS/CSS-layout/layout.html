    # 九宫格布局

    # 1. 方法一(calc)

    <template>
  <div class="parent">
    <div class="item">1</div>
    <div class="item">2</div>
    <div class="item">3</div>
    <div class="item">4</div>
    <div class="item">5</div>
    <div class="item">6</div>
    <div class="item">7</div>
    <div class="item">8</div>
    <div class="item">9</div>
  </div>
</template>
<style>
  * {
    margin: 0;
    padding: 0;
  }
  .parent {
    display: flex;
    flex-wrap: wrap;
    margin-left: -20px;
    margin-top: -20px;
    height: 100%;
  }
  .item {
    /*flex: 0 1 calc((100%-60px)/3); // 用这个代替 width*/
    overflow: hidden;
    width: calc((100% - 60px) / 3);
    padding-bottom: calc((100% - 60px) / 3 - 20px);
    margin-left: 20px;
    margin-top: 20px;
    background: chocolate;
  }
</style>
