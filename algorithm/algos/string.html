    # 最长回文子串 -- 中等

    给你一个字符串 s，找到 s 中最长的回文子串。

    示例 1

    输入： s = "babad"
    输出： "bab"
    解释： "aba" 同样是符合题意的答案。

    示例 2

    输入： s = "cbbd"
    输出： "bb"

    示例 3

    输入： s = "a"
    输出： "a"

    示例 4

    输入： s = "ac"
    输出： "a"

    解法一：双指针-中心扩展法

    /**
 * @param {string} s
 * @return {string}
 */
 // 双指针 中心扩展法 
var longestPalindrome = function(s) {
  if (s.length === 1) return s;
  let maxLen = 0, maxStr = ''
  for (let i=0; i<s.length; i++) {
    // 回文子串是奇数
    let str1 = palindrome(s, i, i)
    // 回文子串是偶数
    let str2 = palindrome(s, i, i + 1)
    // 判断是否大于前面的回文子串
    if (str1.length > maxLen) {
      maxStr = str1;
      maxLen = str1.length;
    }
    // 判断是否大于前面的回文子串
    if (str2.length > maxLen) {
      maxStr = str2;
      maxLen = str2.length;
    }
  }
  return maxStr;
  function palindrome(s, l, r) {
    // 中心扩展法寻找回文子串
    while(l >= 0 && r < s.length && s[l] === s[r]) {
      l--;
      r++
    }
    return s.slice(l + 1, r);
  }
};

    # 回文子串

    给定一个字符串，你的任务是计算这个字符串中有多少个回文子串。

    具有不同开始位置或结束位置的子串，即使是由相同的字符组成，也会被视作不同的子串。

    示例 1

    输入： "abc"
    输出： 3
    解释： 三个回文子串: "a", "b", "c"

    示例 2

    输入： "aaa"
    输出： 6
    解释： 6个回文子串: "a", "a", "a", "aa", "aa", "aaa"

    解法一：双指针

    /**
 * @param {string} s
 * @return {number}
 */
const countSubstrings = (s) => {
  const len = s.length;
  let count = 0;

  for(let i = 0; i < 2 * len - 1; i++) {
    let left = Math.floor(i/2);
    let right = left + i % 2;

    while(left >= 0 && right < len && s[left] === s[right]) {
      count++;
      left--;
      right++;
    }
  }
  return count;
}

    # 最长公共前缀 -- 简单

    编写一个函数来查找字符串数组中的最长公共前缀。

    如果不存在公共前缀，返回空字符串 ""。

    示例 1

    输入： strs = ["flower", "flow", "flight"]
    输出： "fl"

    示例 2

    输入： strs = ["dog", "racecar", "car"]
    输出： "" 解释： 输入不存在公共前缀。

    解法一：双指针

    /**
 * @param {string[]} strs
 * @return {string}
 */
var longestCommonPrefix = function(strs) {
  // 取出第一位
  let first = strs[0];
  // 如果第一位是为空，则直接返回空
  if (first.length === 0) return '';
  // 用来存储最长公共前缀
  let minLen = Number.MAX_FAFE_INTEGER;
  // 循环遍历 strs
  for (let i=1; i<strs.length; i++) {
    // 判断第一个和后面的最长公共前缀
    let cnt = twoStringLongestPrefix(first, strs[i]);
    // 最后提取最长公共前缀，即取最小值
    minLen = Math.min(cnt, minLen);
  }
  // 提取出最长公共前缀部分
  return first.slice(0, minLen);
  // 比较两个数的最长公共前缀
  function twoStringLongestPrefix(first, second) {
    let l = 0, r = 0, cnt = 0;
    while(l < first.length && r < second.length) {
      // 满足条件就将 cnt++，否则直接返回cnt
      if (first[l] === second[r]) {
        cnt++
      } else {
        return cnt;
      }
      l++;
      r++;
    }
    return cnt;
  }
};

    # 无重复字符的最长子串 -- 中等

    给定一个字符串，请你找出其中不含有重复字符的 最长子串 的长度。

    示例 1

    输入: s = "abcabcbb"
    输出: 3
    解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。

    示例 2

    输入: s = "bbbbb"
    输出: 1
    解释: 因为无重复字符的最长子串是 "b"，所以其长度为 1。

    示例 3

    输入: s = "pwwkew"
    输出: 3
    解释: 因为无重复字符的最长子串是"wke"，所以其长度为 3。
    请注意，你的答案必须是 子串 的长度，"pwke"是一个子序列，不是子串。

    示例 4

    输入: s = ""
    输出: 0

    解法一：双指针

    /**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function(s) {
  // obj 用来储存所有字符出现的次数
  let obj = {};
  // 设置左右两个指针，r 用来循环， l 用来循环两个相同字符间的距离
  let l = 0, r = 0;
  // 存储不同字符的大小
  let maxLen = 0;
  while(r < s.length) {
    // 取出当前字符
    let c = s[r];
    // 循环下一个
    r++;
    // 如果obj里有相同字符就加1，否则设置为1
    if (obj[c]) obj[c]++;
    else obj[c] = 1;
    // 如果obj[c]>1，则循环直到找到相同字符
    while(obj[c] > 1) {
      let d = s[l];
      l++;
      obj[d]--;
    }
    // 判断当前字符串长度是否大于前一个字符串长度
    if (maxLen < (r - l)) {
      maxLen = r - l;
    }
  }
  return maxLen;
};

    # 最小覆盖子串 -- 困难

    给你一个字符串 s 、一个字符串 t 。返回 s 中涵盖 t 所有字符的最小子串。如果 s 中不存在涵盖 t 所有字符的子串，则返回空字符串 "" 。

    注意：如果 s 中存在这样的子串，我们保证它是唯一的答案。

    示例 1

    输入： s = "ADOBECODEBANC", t = "ABC"
    输出： "BANC"

    示例 2

    输入： s = "a", t = "a"
    输出： "a"

    解法一：双指针

    /**
 * @param {string} s
 * @param {string} t
 * @return {string}
 */
var minWindow = function(s, t) {
  let need = {}, window = {};
  for (let c of t) {
    if (!need[c]) need[c] = 1;
    else need[c]++;
  }
  let left = 0, right = 0;
  let valid = 0, len = Object.keys(need).length;
  let minLen = s.length + 1, minStr = '';
  while (right < s.length) {
    const d = s[right];
    right++;
    if (!window[d]) window[d] = 1;
    else window[d]++;
    if (need[d] && need[d] === window[d]) {
      valid++;
    }
    while (valid === len) {
      if (right - left < minLen) {
        minLen = right - left;
        minStr = s.slice(left, right);
      }
      let c = s[left];
      left++;
      window[c]--;
      if (need[c] && window[c] < need[c]) {
        valid--;
      }
    }
  }
  return minStr;
};

    # 第一个只出现一次的字符 -- 简单

    在字符串 s 中找出第一个只出现一次的字符。如果没有，返回一个单空格。 s 只包含小写字母。

    示例 1

    s = "abaccdeff"
    返回 "b"

    s = ""
    返回 " "

    解法一：Map 双重循环

    /**
 * @param {string} s
 * @return {character}
 */
const firstUniqChar = str => {
  // 判断边界条件
  if (str.length === 0) return ' ';
  // 设置Map，用来存储字符出现的情况
  let map = new Map();
  // 遍历存储字符的情况
  for (const s of str) {
    map.set(s, (map.get(s) || 0) + 1);
  }
  // 提取第一个只出现一次的字符
  for (const s of map) {
    if (s[1] === 1) return s[0];
  }
  // 否则 返回空
  return ' ';
};

    # 字符串的排列 -- 中等

    输入一个字符串，打印出该字符串中字符的所有排列。

    你可以以任意顺序返回这个字符串数组，但里面不能有重复元素。

    示例 1

    输入： s = "abc"
    输出： ["abc", "acb", "bac", "bca", "cab", "cba"]

    解法一：递归

    const permutation = s => {
  if (s.length === 0) return [''];
  // 递归出口
  if (s.length === 1) return [s];
  const res = [];
  const len = s.length;
  for (let i = 0; i < len; i++) {
    // 取出一个字符为char
    const char = s[i];
    // newStr=去掉char后剩下的字符
    let newStr = s.slice(0, i) + s.slice(i + 1);
    // 递归产生newStr的全排列
    const next = permutation(newStr);
    // 将char与newStr的全排列拼接，放入res
    next.forEach(item => {
        res.push(char + item);
    });
  }
  // 去重
  return [...new Set(res)];
};

    # 有效的括号 -- 简单

    给定一个只包括 '('，')'，'{'，'}'，'['，']'的字符串 s，判断字符串是否有效。

    有效字符串需满足： -左括号必须用相同类型的右括号闭合。

    • 左括号必须以正确的顺序闭合。

    示例 1

    输入： s = "()"
    输出： true

    示例 2

    输入： s = "()[]{}"
    输出： true

    示例 3

    输入： s = "(]"
    输出： false

    示例 4

    输入： s = "([)]"
    输出： false

    示例 5

    输入： s = "{[]}" 输出： true

    解法一：栈

    /*思路 
1.利用array 模拟stack
2.建立一个字典dic来匹配对应的括号
3.因为括号需要正向的所以 一旦遇到 } ] )且stack数组为空 则表示违法
4.如果遇到了任意一个 } ] )则对比stack的顶层元素 不符合那也返回为false，
符合则stack.pop 排除最末尾的那个
5.循环结束后 看stack 是否为空 空则返回ture 反之返回false 
*/

var isValid = function (s) {
  // 创建字典
  const dic = {
    "{": "}",
    "[": "]",
    "(": ")",
  };

  const keys = Object.keys(dic);
  const stack = [];

  for (const element of s) {
    // 因为括号需要正向的所以 一旦遇到 } ] )且stack数组为空 则表示违法
    if (!stack.length && !keys.includes(element)) {
      return false;
    }

    // 如果匹配dic的key说明是左边的括号 放入stack顶部
    if (keys.includes(element)) {
      stack.push(element);
    } else {
      // 匹配顶部元素
      const matchItem = dic[stack.pop()];
      //   不成功则返回false
      if (matchItem !== element) {
        return false;
      }
    }
  }

  //   循环结束后 看stack 是否为空 空则返回ture 反之返回false
  return stack.length > 0 ? false : true;
};

/* 复杂度
时间 最大为O(n) 最小为O(1)
空间 最大为O(n/2) 最小为O(1)
*/

    # 括号生成 -- 中等

    数字 n 代表生成括号的对数，请你设计一个函数，用于能够生成所有可能的并且 有效的 括号组合。

    有效括号组合需满足：左括号必须以正确的顺序闭合。

    示例 1

    输入： n = 3 输出： ["((()))","(()())","(())()","()(())","()()()"]

    示例 2

    输入： n = 输出： ["()"]

    解法一：DFS

    /**
 * @param {number} n
 * @return {string[]}
 */
var generateParenthesis = function(n) {
  const res = []
  // 左边括号不能大于 n, 右边括号不能大于左边括号，否则返回
  const dfs = (path, open, close) => {
    if (open > n || close > open) return
    // path 长度为 n 的两倍表示已经匹配完成，则加到 res 里
    if (path.length === 2 * n) {
      res.push(path)
      return
    }
    // 进行 DFS 递归
    dfs(path + "(", open + 1, close)
    dfs(path + ")", open, close + 1)
  }
  dfs("", 0, 0)
  return res
};

    # 单词拆分 -- 中等

    给定一个非空字符串 s 和一个包含非空单词的列表 wordDict，判定s 是否可以被空格拆分为一个或多个在字典中出现的单词。

    说明：

    • 拆分时可以重复使用字典中的单词。
    • 你可以假设字典中没有重复的单词。

    示例 1

    输入： s = "leetcode", wordDict = ["leet", "code"]
    输出： true
    解析： 返回 true 因为 "leetcode" 可以被拆分成 "leet code"。

    示例 2

    输入： s = "applepenapple", wordDict = ["apple", "pen"]
    输出： true
    解析： 返回 true 因为 "applepenapple" 可以被拆分成 "apple pen apple"。 注意你可以重复使用字典中的单词。

    示例 1

    输入： s = "catsandog", wordDict = ["cats", "dog", "sand", "and", "cat"]
    输出： false

    解法一：DFS+记忆化搜索

    /**
 * @param {string} s
 * @param {string[]} wordDict
 * @return {boolean}
 */
var wordBreak = function(s, wordDict) {
  if (!s || !s.length || !wordDict) return false;
  const wordSet = new Set(wordDict);
  const memo = new Map(); // {startIndex: false}
  function canBreak(start) {
    if (memo.has(start)) return memo.get(start);
    if (start === s.length) return true;
    for (let i = start; i < s.length; ++i) {
      const temp = s.substring(start, i + 1);
      if (wordSet.has(temp) && canBreak(i + 1)) {
        memo.set(start, true);
        return true;
      }
    }
    memo.set(start, false);
    return false;
  }
  return canBreak(0);
};
